## 📁 Part 1: File Handling

### Opening and Reading Files

Python uses the `open()` function to work with files. The **best practice** is to use `with` (context manager) — it automatically closes the file when you're done.

#### Reading an Entire File

```python
# ✅ Best practice: use "with" — file is automatically closed
with open("/etc/hostname", "r") as f:
    content = f.read()
    print(content)

# What "r" means — it's the MODE:
# "r"  = Read (default) — file must exist
# "w"  = Write — creates file or OVERWRITES existing
# "a"  = Append — adds to the end of existing file
# "x"  = Exclusive create — creates file, fails if it already exists
# "rb" = Read Binary (for images, PDFs, etc.)
# "wb" = Write Binary
```

#### Reading Line by Line

```python
# Method 1: readlines() — returns a list of all lines
with open("/etc/hosts", "r") as f:
    lines = f.readlines()
    for line in lines:
        print(line.strip())    # strip() removes the newline character

# Method 2: readline() — reads one line at a time
with open("/etc/hosts", "r") as f:
    first_line = f.readline()
    second_line = f.readline()
    print(f"First: {first_line.strip()}")
    print(f"Second: {second_line.strip()}")

# Method 3: Iterate directly — most memory efficient!
with open("/etc/hosts", "r") as f:
    for line in f:
        print(line.strip())
```

**When to use which?**

| Method | When to Use | Memory |
|--------|------------|--------|
| `f.read()` | Small files, need entire content | Loads ALL into memory |
| `f.readlines()` | Need all lines as a list | Loads ALL into memory |
| `f.readline()` | Need specific lines (1st, 2nd...) | One line at a time |
| `for line in f:` | Large files, process line by line | One line at a time ✅ |

#### Real DevOps Example: Parse a Log File for Errors

```python
#!/usr/bin/env python3
"""Parse an Nginx log file and extract error entries."""

log_file = "/var/log/nginx/error.log"
errors = []
warnings = []

try:
    with open(log_file, "r") as f:
        for line_num, line in enumerate(f, start=1):
            line = line.strip()
            if "error" in line.lower():
                errors.append({"line": line_num, "content": line})
            elif "warn" in line.lower():
                warnings.append({"line": line_num, "content": line})

    print(f"📊 Log Analysis: {log_file}")
    print(f"   Errors:   {len(errors)}")
    print(f"   Warnings: {len(warnings)}")

    if errors:
        print("\n🔥 Last 5 Errors:")
        for e in errors[-5:]:
            print(f"   Line {e['line']}: {e['content'][:100]}")

except FileNotFoundError:
    print(f"❌ Log file not found: {log_file}")
except PermissionError:
    print(f"❌ No permission to read: {log_file}")
```

### Working with JSON Files

JSON is everywhere in DevOps — API responses, configs, Terraform state, Kubernetes manifests...

```python
import json

# === Writing JSON ===
server_data = {
    "servers": [
        {"name": "web01", "ip": "10.0.1.10", "status": "running"},
        {"name": "web02", "ip": "10.0.1.11", "status": "running"},
        {"name": "db01",  "ip": "10.0.2.10", "status": "stopped"}
    ],
    "metadata": {
        "total": 3,
        "last_updated": "2026-05-29T10:30:00Z"
    }
}

# Write to file (indent=4 makes it pretty-printed)
with open("/tmp/inventory.json", "w") as f:
    json.dump(server_data, f, indent=4)

print("✅ JSON written!")

# === Reading JSON ===
with open("/tmp/inventory.json", "r") as f:
    data = json.load(f)

# Access the data
print(f"\nTotal servers: {data['metadata']['total']}")
for server in data["servers"]:
    status = "✅" if server["status"] == "running" else "❌"
    print(f"  {status} {server['name']} — {server['ip']}")

# === String ↔ JSON (for API responses) ===
# Python dict → JSON string
json_string = json.dumps(server_data, indent=2)
print(type(json_string))    # <class 'str'>

# JSON string → Python dict
parsed = json.loads(json_string)
print(type(parsed))    # <class 'dict'>
```

### Quick Reference — `json` module

| Function | What It Does | Input → Output |
|----------|-------------|----------------|
| `json.dump(data, file)` | Write dict to a JSON file | dict → file |
| `json.load(file)` | Read JSON file into dict | file → dict |
| `json.dumps(data)` | Convert dict to JSON string | dict → str |
| `json.loads(string)` | Convert JSON string to dict | str → dict |

---
```python
name: db-deployment
```

### Real DevOps Example: Directory Cleanup Script

```python
#!/usr/bin/env python3
"""Clean up old files from a directory based on age."""

import os
import time

def cleanup_old_files(directory, max_age_days=7, dry_run=True):
    """Remove files older than max_age_days from directory."""
    
    if not os.path.isdir(directory):
        print(f"❌ Directory not found: {directory}")
        return
    
    now = time.time()
    max_age_seconds = max_age_days * 24 * 60 * 60
    
    removed_count = 0
    removed_size = 0
    
    print(f"🧹 Cleaning up: {directory}")
    print(f"   Max age: {max_age_days} days")
    print(f"   Mode: {'DRY RUN' if dry_run else 'LIVE'}")
    print()
    
    for root, dirs, files in os.walk(directory):
        for filename in files:
            filepath = os.path.join(root, filename)
            
            try:
                file_age = now - os.path.getmtime(filepath)
                file_size = os.path.getsize(filepath)
                
                if file_age > max_age_seconds:
                    age_days = int(file_age / (24 * 60 * 60))
                    size_mb = file_size / (1024 * 1024)
                    
                    if dry_run:
                        print(f"  🔍 Would delete: {filepath} "
                              f"({age_days} days old, {size_mb:.2f} MB)")
                    else:
                        os.remove(filepath)
                        print(f"  🗑️ Deleted: {filepath}")
                    
                    removed_count += 1
                    removed_size += file_size
                    
            except (OSError, PermissionError) as e:
                print(f"  ⚠️ Error with {filepath}: {e}")
    
    total_mb = removed_size / (1024 * 1024)
    action = "Would remove" if dry_run else "Removed"
    print(f"\n📊 {action} {removed_count} files ({total_mb:.2f} MB)")

# Run it (dry_run=True means it won't actually delete anything)
cleanup_old_files("/tmp/logs", max_age_days=7, dry_run=True)

# When you're sure it's correct, set dry_run=False
# cleanup_old_files("/tmp/logs", max_age_days=7, dry_run=False)
```

## 🐚 Part 3: The `shutil` Module — High-Level File Operations

The `os` module does basic file operations. For copying, moving, and archiving, use `shutil` (shell utilities).

```python
import shutil

# Copy a file
shutil.copy("/tmp/source.txt", "/tmp/destination.txt")        # Copy file
shutil.copy2("/tmp/source.txt", "/tmp/destination.txt")       # Copy + preserve metadata

# Copy an entire directory tree
shutil.copytree("/tmp/myapp", "/tmp/myapp_backup")

# Move a file or directory (like mv)
shutil.move("/tmp/old_location.txt", "/tmp/new_location.txt")

# Remove an entire directory tree (like rm -rf) — BE CAREFUL!
shutil.rmtree("/tmp/myapp_backup")

# Get disk usage
total, used, free = shutil.disk_usage("/")
print(f"Total: {total // (1024**3)} GB")
print(f"Used:  {used // (1024**3)} GB")
print(f"Free:  {free // (1024**3)} GB")

# Create an archive
shutil.make_archive("/tmp/backup_2026", "gztar", "/var/www/html")
# Creates /tmp/backup_2026.tar.gz containing everything in /var/www/html

# Extract an archive
shutil.unpack_archive("/tmp/backup_2026.tar.gz", "/tmp/restored")
```

### DevOps Example: Backup Script with shutil

```python
#!/usr/bin/env python3
"""Create a timestamped backup of a directory."""

import shutil
import os
from datetime import datetime

def create_backup(source_dir, backup_dir="/backups"):
    """Create a timestamped backup of source_dir."""
    
    if not os.path.isdir(source_dir):
        print(f"❌ Source directory not found: {source_dir}")
        return None
    
    # Create backup directory if needed
    os.makedirs(backup_dir, exist_ok=True)
    
    # Generate timestamped backup name
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    dir_name = os.path.basename(source_dir.rstrip("/"))
    backup_name = f"{dir_name}_backup_{timestamp}"
    backup_path = os.path.join(backup_dir, backup_name)
    
    print(f"📦 Creating backup...")
    print(f"   Source: {source_dir}")
    print(f"   Backup: {backup_path}.tar.gz")
    
    # Create compressed archive
    archive_path = shutil.make_archive(backup_path, "gztar", source_dir)
    
    # Get size
    size_mb = os.path.getsize(archive_path) / (1024 * 1024)
    print(f"✅ Backup created! Size: {size_mb:.2f} MB")
    
    return archive_path

# Usage
create_backup("/var/www/html")
```

## ⚡ Part 4: The `subprocess` Module — Running Shell Commands

This is where it gets **really powerful**. The `subprocess` module lets you run any shell command from Python and capture its output.

### Why `subprocess` and Not `os.system()`?

```python
# ❌ Old way — os.system() — DON'T USE THIS
import os
os.system("ls -la")    # No way to capture output, security risks

# ✅ Modern way — subprocess.run()
import subprocess
result = subprocess.run(["ls", "-la"], capture_output=True, text=True)
print(result.stdout)
```

Why `subprocess.run()` is better:
- Capture stdout and stderr
- Check return codes
- Handle errors gracefully
- More secure (avoids shell injection)

### Basic Usage — `subprocess.run()`

```python
import subprocess

# Run a simple command
result = subprocess.run(["echo", "Hello from Python!"], capture_output=True, text=True)

print(f"Output: {result.stdout}")
print(f"Errors: {result.stderr}")
print(f"Return code: {result.returncode}")
```

**Breaking it down:**

| Parameter | What It Does |
|-----------|-------------|
| `["echo", "Hello from Python!"]` | The command as a **list** (each argument is a separate item) |
| `capture_output=True` | Capture stdout and stderr (otherwise they go to the terminal) |
| `text=True` | Return output as string (not bytes) |

### Running Various Commands

```python
import subprocess

# Check disk usage
result = subprocess.run(["df", "-h"], capture_output=True, text=True)
print("💾 Disk Usage:")
print(result.stdout)

# Check memory
result = subprocess.run(["free", "-h"], capture_output=True, text=True)

# Get hostname
```

### Using `shell=True` — When You Need Pipes and Redirects

Sometimes you need shell features like pipes (`|`), redirects (`>`), or wildcards (`*`):

```python
import subprocess

# ⚠️ shell=True lets you use pipes, but be careful with user input!
result = subprocess.run(
    "ps aux | head -5",
    shell=True,
    capture_output=True,
    text=True
)
print(result.stdout)

# Count files in a directory
result = subprocess.run(
    "find /var/log -name '*.log' | wc -l",
    shell=True,
    capture_output=True,
    text=True
)
print(f"Log files found: {result.stdout.strip()}")
```

> **⚠️ Security Warning:** NEVER use `shell=True` with user-provided input! It can lead to **command injection attacks**. Always use the list format for commands when dealing with external input.

```python
# ❌ DANGEROUS — Shell injection vulnerability!
user_input = "file.txt; rm -rf /"    # Malicious input
subprocess.run(f"cat {user_input}", shell=True)    # This would be catastrophic!

# ✅ SAFE — List format prevents injection
user_input = "file.txt; rm -rf /"
subprocess.run(["cat", user_input], capture_output=True, text=True)
# This safely treats the entire string as a single filename argument
```

### Handling Errors

```python
import subprocess

# Method 1: Check returncode manually
result = subprocess.run(["ls", "/nonexistent"], capture_output=True, text=True)
if result.returncode != 0:
    print(f"❌ Command failed (exit code {result.returncode})")
    print(f"   Error: {result.stderr.strip()}")
else:
    print(f"✅ Output: {result.stdout}")

# Method 2: Use check=True — raises exception on failure
try:
    result = subprocess.run(
        ["systemctl", "status", "nginx"],
        capture_output=True,
        text=True,
        check=True    # Raises CalledProcessError if return code != 0
    )
    print(f"✅ Nginx is running:\n{result.stdout}")
except subprocess.CalledProcessError as e:
    print(f"❌ Command failed with exit code {e.returncode}")
    print(f"   Error: {e.stderr}")
```

### Getting Live Output (Streaming)

Sometimes you want to see the output as it happens (not after the command finishes):

```python
import subprocess

# Stream output in real-time
process = subprocess.Popen(
    ["ping", "-c", "5", "google.com"],
    stdout=subprocess.PIPE,
    stderr=subprocess.PIPE,
    text=True
)

# Read output line by line as it comes
for line in process.stdout:
    print(f"📡 {line.strip()}")

# Wait for the process to finish
process.wait()
print(f"\nExit code: {process.returncode}")
```

### `subprocess.run()` vs `subprocess.Popen()`

| Feature | `subprocess.run()` | `subprocess.Popen()` |
|---------|--------------------|-----------------------|
| Use case | Simple, run-and-wait | Advanced, streaming, async |
| Blocking | Yes — waits for completion | No — runs in background |
| Output | Available after command finishes | Available in real-time |
| Complexity | Simple | More complex |
| When to use | 95% of the time | When you need real-time output or process control |

> **Rule:** Start with `subprocess.run()`. Only use `subprocess.Popen()` when you actually need real-time output or background processes.

```python
#!/usr/bin/env python3
"""Gather comprehensive system information and save to a report."""

import os
import subprocess
import json
from datetime import datetime
import platform
import shutil


def run_command(cmd, shell=False):
    """Run a command and return its output."""
    try:
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            timeout=30,
            shell=shell
        )
        return result.stdout.strip() if result.returncode == 0 else f"Error: {result.stderr.strip()}"
    except (subprocess.TimeoutExpired, FileNotFoundError) as e:
        return f"Error: {e}"


def gather_system_info():
    """Collect all system information."""
    info = {
        "timestamp": datetime.now().isoformat(),
        "hostname": platform.node(),
        "os": {
            "system": platform.system(),
            "release": platform.release(),
            "version": platform.version(),
            "architecture": platform.machine(),
        },
        "python_version": platform.python_version(),
    }

    # Disk usage
    disk = shutil.disk_usage("/")
    info["disk"] = {
        "total_gb": round(disk.total / (1024**3), 2),
        "used_gb": round(disk.used / (1024**3), 2),
        "free_gb": round(disk.free / (1024**3), 2),
        "usage_percent": round((disk.used / disk.total) * 100, 1),
    }

    # Uptime
    info["uptime"] = run_command(["uptime", "-p"])

    # Memory
    info["memory"] = run_command(["free", "-h"])

    # CPU info
    info["cpu_count"] = os.cpu_count()

    # Network interfaces
    info["network"] = run_command("ip -brief addr show", shell=True)

    # Running services
    info["docker_status"] = run_command(["systemctl", "is-active", "docker"])

    # Environment
    info["environment"] = {
        "USER": os.environ.get("USER", "unknown"),
        "HOME": os.environ.get("HOME", "unknown"),
        "SHELL": os.environ.get("SHELL", "unknown"),
    }

    return info


def print_report(info):
    """Print a formatted system report."""
    print("=" * 60)
    print("    🖥️  SYSTEM INFORMATION REPORT")
    print(f"    Generated: {info['timestamp']}")
    print("=" * 60)

    print(f"\n📌 Hostname: {info['hostname']}")
    print(f"📌 OS: {info['os']['system']} {info['os']['release']}")
    print(f"📌 Architecture: {info['os']['architecture']}")
    print(f"📌 Python: {info['python_version']}")
    print(f"📌 CPUs: {info['cpu_count']}")
    print(f"📌 Uptime: {info['uptime']}")

    disk = info["disk"]
    print(f"\n💾 Disk Usage:")
    print(f"   Total: {disk['total_gb']} GB")
    print(f"   Used:  {disk['used_gb']} GB ({disk['usage_percent']}%)")
    print(f"   Free:  {disk['free_gb']} GB")

    print(f"\n🧠 Memory:\n{info['memory']}")
    print(f"\n🌐 Network:\n{info['network']}")
    print(f"\n🐳 Docker: {info['docker_status']}")

    print("\n" + "=" * 60)


# === Main ===
info = gather_system_info()
print_report(info)

# Save to JSON file
report_file = f"/tmp/system_report_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
with open(report_file, "w") as f:
    json.dump(info, f, indent=4)
print(f"\n💾 Report saved to: {report_file}")
```

### Script 2: Multi-Server Health Checker

```python
#!/usr/bin/env python3
"""Check the health of multiple servers using ping and port checks."""

import subprocess
import json
import os
from datetime import datetime


def ping_host(host, count=2, timeout=5):
    """Ping a host and return True if reachable."""
    try:
        result = subprocess.run(
            ["ping", "-c", str(count), "-W", str(timeout), host],
            capture_output=True,
            text=True,
            timeout=timeout + 5
        )
        return result.returncode == 0
    except (subprocess.TimeoutExpired, FileNotFoundError):
        return False


def check_port(host, port, timeout=3):
    """Check if a TCP port is open on a host."""
    try:
        result = subprocess.run(
            ["bash", "-c", f"echo > /dev/tcp/{host}/{port}"],
            capture_output=True,
            text=True,
            timeout=timeout
        )
        return result.returncode == 0
    except (subprocess.TimeoutExpired, FileNotFoundError):
        return False


def check_servers(servers):
    """Check health of all servers."""
    results = []

    for server in servers:
        name = server["name"]
        host = server["host"]
        ports = server.get("ports", [])

        print(f"\n🔍 Checking {name} ({host})...")

        # Ping check
        is_reachable = ping_host(host)
        status = "✅ Reachable" if is_reachable else "❌ Unreachable"
        print(f"   Ping: {status}")

        # Port checks
        port_results = {}
        for port in ports:
            is_open = check_port(host, port)
            port_status = "✅ Open" if is_open else "❌ Closed"
            print(f"   Port {port}: {port_status}")
            port_results[port] = is_open

        results.append({
            "name": name,
            "host": host,
            "reachable": is_reachable,
            "ports": port_results,
            "checked_at": datetime.now().isoformat()
        })

    return results


# === Server Inventory ===
servers = [
    {"name": "Web Server",      "host": "10.0.1.10", "ports": [80, 443]},
    {"name": "Database Server",  "host": "10.0.2.10", "ports": [5432]},
    {"name": "Redis Cache",      "host": "10.0.3.10", "ports": [6379]},
    {"name": "Google DNS",       "host": "8.8.8.8",   "ports": [53]},
]

print("=" * 50)
print("    🏥 SERVER HEALTH CHECK")
print("=" * 50)

results = check_servers(servers)

# Summary
print("\n" + "=" * 50)
print("📊 SUMMARY")
print("=" * 50)

total = len(results)
healthy = sum(1 for r in results if r["reachable"])
unhealthy = total - healthy

print(f"   Total:     {total}")
print(f"   Healthy:   {healthy} ✅")
print(f"   Unhealthy: {unhealthy} ❌")

# Save results
report_file = "/tmp/health_check.json"
with open(report_file, "w") as f:
    json.dump(results, f, indent=4)
print(f"\n💾 Report saved: {report_file}")
```

### Script 3: Automated Deployment Script

```python
#!/usr/bin/env python3
"""Simple automated deployment script using subprocess."""

import subprocess
import os
import sys
from datetime import datetime


def run(cmd, description, cwd=None, check=True):
    """Run a command with nice output formatting."""
    print(f"\n🔷 {description}")
    print(f"   $ {cmd if isinstance(cmd, str) else ' '.join(cmd)}")

    try:
        result = subprocess.run(
            cmd,
            shell=isinstance(cmd, str),
            capture_output=True,
            text=True,
            cwd=cwd,
            timeout=300
        )

        if result.stdout.strip():
            for line in result.stdout.strip().split("\n")[:10]:
                print(f"   {line}")

        if result.returncode != 0:
            print(f"   ❌ FAILED (exit code {result.returncode})")
            if result.stderr.strip():
                print(f"   Error: {result.stderr.strip()}")
            if check:
                sys.exit(1)
            return False

        print(f"   ✅ Done")
        return True

    except subprocess.TimeoutExpired:
        print(f"   ⏰ TIMED OUT")
        if check:
            sys.exit(1)
        return False


def deploy(app_dir, branch="main"):
    """Run the deployment steps."""

    print("=" * 60)
    print("    🚀 DEPLOYMENT STARTED")
    print(f"    App: {app_dir}")
    print(f"    Branch: {branch}")
    print(f"    Time: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    print("=" * 60)

    # Step 1: Pull latest code
    run(["git", "fetch", "--all"],
        "Fetching latest code", cwd=app_dir)

    run(["git", "checkout", branch],
        f"Switching to branch: {branch}", cwd=app_dir)

    run(["git", "pull", "origin", branch],
        "Pulling latest changes", cwd=app_dir)

    # Step 2: Install dependencies
    if os.path.exists(os.path.join(app_dir, "requirements.txt")):
        run(["pip3", "install", "-r", "requirements.txt"],
            "Installing Python dependencies", cwd=app_dir)

    if os.path.exists(os.path.join(app_dir, "package.json")):
        run(["npm", "install"],
            "Installing Node.js dependencies", cwd=app_dir)

    # Step 3: Run tests
    run(["python3", "-m", "pytest", "--tb=short"],
        "Running tests", cwd=app_dir, check=False)

    # Step 4: Restart the application
    run(["sudo", "systemctl", "restart", "myapp"],
        "Restarting application")

    # Step 5: Verify health
    run("curl -sf http://localhost:8080/health || exit 1",
        "Health check")

    print("\n" + "=" * 60)
    print("    ✅ DEPLOYMENT COMPLETED SUCCESSFULLY!")
    print("=" * 60)


# Run deployment
if __name__ == "__main__":
    app_directory = sys.argv[1] if len(sys.argv) > 1 else "/var/www/myapp"
    deploy_branch = sys.argv[2] if len(sys.argv) > 2 else "main"
    deploy(app_directory, deploy_branch)
```

## 📁 The `pathlib` Module — Modern Path Handling

`pathlib` is a modern, object-oriented way to work with file paths (Python 3.4+). Many prefer it over `os.path`:

```python
from pathlib import Path

# Create Path objects
home = Path.home()
config = Path("/etc/nginx/nginx.conf")
log_dir = Path("/var/log")

# Path operations
print(config.name)        # nginx.conf
print(config.stem)        # nginx
print(config.suffix)      # .conf
print(config.parent)      # /etc/nginx
print(config.exists())    # True or False
print(config.is_file())   # True
print(log_dir.is_dir())   # True

# Join paths — use / operator!
backup_dir = Path("/backups") / "2026" / "05"
print(backup_dir)    # /backups/2026/05

# Create directories
backup_dir.mkdir(parents=True, exist_ok=True)

# List files
for f in Path("/var/log").iterdir():
    print(f"  {'📁' if f.is_dir() else '📄'} {f.name}")

# Find files with glob
for f in Path("/var/log").glob("*.log"):
    print(f"  📄 {f.name} ({f.stat().st_size} bytes)")

# Recursive glob
for f in Path("/etc").rglob("*.conf"):
    print(f"  📄 {f}")

# Read and write files (super clean!)
content = Path("/etc/hostname").read_text()
print(f"Hostname: {content.strip()}")

Path("/tmp/hello.txt").write_text("Hello from pathlib!")
```

## 📝 Quick Reference Table

### File Handling

| Operation | Code |
|-----------|------|
| Read file | `with open(f, "r") as fh: data = fh.read()` |
| Write file | `with open(f, "w") as fh: fh.write(data)` |
| Append | `with open(f, "a") as fh: fh.write(data)` |
| Read lines | `with open(f) as fh: lines = fh.readlines()` |
| Read JSON | `data = json.load(open(f))` |
| Write JSON | `json.dump(data, open(f, "w"), indent=4)` |
| Read YAML | `data = yaml.safe_load(open(f))` |

### os Module

| Operation | Code |
|-----------|------|
| Current dir | `os.getcwd()` |
| List dir | `os.listdir(path)` |
| File exists? | `os.path.exists(path)` |
| Is file? | `os.path.isfile(path)` |
| Is dir? | `os.path.isdir(path)` |
| Create dir | `os.makedirs(path, exist_ok=True)` |
| Remove file | `os.remove(path)` |
| Env variable | `os.environ.get("KEY", "default")` |
| Join paths | `os.path.join("/var", "log", "app.log")` |
| Walk tree | `for root, dirs, files in os.walk(path):` |

### subprocess Module

| Operation | Code |
|-----------|------|
| Run command | `subprocess.run(["cmd", "arg"], capture_output=True, text=True)` |
| With shell | `subprocess.run("cmd | cmd2", shell=True, capture_output=True, text=True)` |
| Check errors | `subprocess.run([...], check=True)` |
| Timeout | `subprocess.run([...], timeout=30)` |
| Custom dir | `subprocess.run([...], cwd="/path")` |
