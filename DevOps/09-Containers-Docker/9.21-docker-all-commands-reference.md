### Using Volumes with `docker run`

```bash
# Named volume
docker run -v mydata:/app/data nginx

# Anonymous volume
docker run -v /app/data nginx

# Bind mount (host path)
docker run -v /host/path:/container/path nginx

# Read-only volume
docker run -v mydata:/app/data:ro nginx

# Using --mount (more explicit, preferred)
docker run --mount type=volume,source=mydata,target=/app/data nginx
docker run --mount type=bind,source=/host/path,target=/container/path nginx
docker run --mount type=tmpfs,target=/app/tmp,tmpfs-size=100m nginx

# Volume from another container
docker run --volumes-from other_container nginx
```

## Quick Reference: Formatting Output

Docker uses Go templates for formatting:

```bash
# List container IDs only
docker ps -q

# Custom format
docker ps --format "table {{.ID}}\t{{.Names}}\t{{.Status}}"

# JSON output
docker ps --format json

# Inspect specific field
docker inspect --format '{{.State.Status}}' mycontainer
docker inspect --format '{{.NetworkSettings.IPAddress}}' mycontainer
docker inspect --format '{{json .Config.Env}}' mycontainer

# Filter containers
docker ps --filter "status=running"
docker ps --filter "name=web"
docker ps --filter "label=env=prod"
docker ps --filter "ancestor=nginx"
docker ps --filter "network=mynet"
docker ps --filter "health=healthy"

# Filter images
docker images --filter "dangling=true"
docker images --filter "reference=nginx*"
docker images --filter "before=nginx:1.26"
docker images --filter "since=nginx:1.25"
docker images --filter "label=maintainer=me"
```

## Quick Reference: Common Patterns

```bash
# Remove all stopped containers
docker rm $(docker ps -aq -f status=exited)

# Remove all images
docker rmi $(docker images -q)

# Stop all running containers
docker stop $(docker ps -q)

# Kill all running containers
docker kill $(docker ps -q)

# Remove all containers (force)
docker rm -f $(docker ps -aq)

# Remove dangling images
docker rmi $(docker images -f dangling=true -q)

# Get container IP address
docker inspect -f '{{range.NetworkSettings.Networks}}{{.IPAddress}}{{end}}' mycontainer

# Follow logs with timestamps
docker logs -f --timestamps mycontainer

# Enter running container
docker exec -it mycontainer /bin/bash
docker exec -it mycontainer /bin/sh    # for Alpine/minimal images

# Copy file from container
docker cp mycontainer:/path/to/file ./local-file

# Check container resource usage
docker stats --no-stream

# Export and import container
docker export mycontainer > container.tar
docker import container.tar myimage:tag

# Save and load image
docker save myimage:tag > image.tar
docker load < image.tar

# Tag and push
docker tag myimage:v1 registry.example.com/myimage:v1
docker push registry.example.com/myimage:v1

# Run with all common options
docker run -d \
  --name myapp \
  --restart unless-stopped \
  -p 8080:80 \
  -e NODE_ENV=production \
  -v appdata:/app/data \
  --network mynet \
  --memory 512m \
  --cpus 1.5 \
  --health-cmd "curl -f http://localhost/" \
  --health-interval 30s \
  myimage:v1
```
