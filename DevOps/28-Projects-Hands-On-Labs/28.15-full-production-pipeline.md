# app/main.py
import logging
from flask import Flask
from app.config import Config
from app.routes.health import health_bp
from app.routes.items import items_bp


def create_app():
    """Application factory — creates and configures the Flask app."""
    app = Flask(__name__)

    # Configure logging
    logging.basicConfig(
        level=getattr(logging, Config.LOG_LEVEL),
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s"
    )
    logger = logging.getLogger(__name__)
    logger.info(f"Starting {Config.APP_NAME} v{Config.APP_VERSION} in {Config.APP_ENV} mode")

    # Register blueprints (route groups)
    app.register_blueprint(health_bp)
    app.register_blueprint(items_bp, url_prefix="/api")

    return app


# Entry point
if __name__ == "__main__":
    app = create_app()
    app.run(host="0.0.0.0", port=Config.PORT)
```

**What's happening here:**
- `create_app()` is an **application factory** — a best practice for Flask apps
- **Blueprints** organize routes into modules (health, items)
- Logging is configured using Python's built-in `logging` module
- The app listens on `0.0.0.0` (all interfaces) — required for containers

# app/routes/health.py
import time
from flask import Blueprint, jsonify
from app.config import Config

health_bp = Blueprint("health", __name__)

# Track when the app started
START_TIME = time.time()


@health_bp.route("/health", methods=["GET"])
def health_check():
    """
    Health check endpoint.
    Used by Kubernetes liveness and readiness probes.
    Returns the app's health status.
    """
    uptime = time.time() - START_TIME
    return jsonify({
        "status": "healthy",
        "app_name": Config.APP_NAME,
        "version": Config.APP_VERSION,
        "environment": Config.APP_ENV,
        "uptime_seconds": round(uptime, 2)
    }), 200


@health_bp.route("/ready", methods=["GET"])
def readiness_check():
    """
    Readiness check endpoint.
    Kubernetes uses this to know if the pod can receive traffic.
    In a real app, you'd check DB connectivity, cache, etc.
    """
    # In production, add checks like:
    # - Can we connect to the database?
    # - Is the cache available?
    # - Are dependent services reachable?
    return jsonify({
        "status": "ready",
        "checks": {
            "database": "ok",
            "cache": "ok"
        }
    }), 200
```

**Why two health endpoints?**

| Endpoint | K8s Probe | What It Checks |
|----------|-----------|---------------|
| `/health` | **Liveness** probe | "Is the app alive?" If not, K8s restarts the pod |
| `/ready` | **Readiness** probe | "Can the app serve traffic?" If not, K8s removes it from the Service |

**`app/routes/items.py`:**
```python
# app/routes/items.py
import logging
from flask import Blueprint, jsonify, request

items_bp = Blueprint("items", __name__)
logger = logging.getLogger(__name__)

# In-memory store (in production, this would be a database)
items_store = [
    {"id": 1, "name": "Laptop", "price": 999.99},
    {"id": 2, "name": "Keyboard", "price": 79.99},
    {"id": 3, "name": "Mouse", "price": 29.99},
]


@items_bp.route("/items", methods=["GET"])
def get_items():
    """Get all items."""
    logger.info(f"Fetching all items. Count: {len(items_store)}")
    return jsonify({"items": items_store, "count": len(items_store)}), 200


@items_bp.route("/items/<int:item_id>", methods=["GET"])
def get_item(item_id):
    """Get a single item by ID."""
    item = next((i for i in items_store if i["id"] == item_id), None)
    if not item:
        logger.warning(f"Item {item_id} not found")
        return jsonify({"error": "Item not found"}), 404
    return jsonify(item), 200


@items_bp.route("/items", methods=["POST"])
def create_item():
    """Create a new item."""
    data = request.get_json()
    if not data or "name" not in data or "price" not in data:
        return jsonify({"error": "Missing required fields: name, price"}), 400

    new_item = {
        "id": max(i["id"] for i in items_store) + 1 if items_store else 1,
        "name": str(data["name"]),
        "price": float(data["price"])
    }
    items_store.append(new_item)
    logger.info(f"Created item: {new_item}")
    return jsonify(new_item), 201
```

### Step 1.5 — Dockerfile

```dockerfile
# Dockerfile
# ══════════════════════════════════════════════════════════
# Multi-stage build for a production-ready Python container
# ══════════════════════════════════════════════════════════

# ── Stage 1: Builder ────────────────────────────────────
# Install dependencies in a temporary build stage
FROM python:3.11-slim AS builder

WORKDIR /build

# Copy and install dependencies first (Docker layer caching!)
COPY requirements.txt .
RUN pip install --no-cache-dir --prefix=/install -r requirements.txt

# ── Stage 2: Production ────────────────────────────────
# Final slim image with only what we need
FROM python:3.11-slim AS production

# Security: Don't run as root!
RUN groupadd -r appuser && useradd -r -g appuser -d /app -s /sbin/nologin appuser

WORKDIR /app

# Copy installed packages from builder stage
COPY --from=builder /install /usr/local

# Copy application code
COPY app/ ./app/

# Set ownership
RUN chown -R appuser:appuser /app

# Switch to non-root user
USER appuser

# Expose port
EXPOSE 5000

# Health check — Docker will monitor this
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:5000/health')" || exit 1

# Run with gunicorn (production WSGI server)
# - Workers: 4 (adjust based on CPU cores)
# - Bind: all interfaces on port 5000
CMD ["gunicorn", "--bind", "0.0.0.0:5000", "--workers", "4", "--access-logfile", "-", "app.main:create_app()"]
```

**Why multi-stage build?**
- **Stage 1 (builder):** Installs all dependencies (may need compilers, dev packages)
- **Stage 2 (production):** Only copies the installed packages — no build tools in final image
- Result: smaller, more secure image

**Security best practices in this Dockerfile:**

| Practice | Why |
|----------|-----|
| `python:3.11-slim` | Minimal base image (less attack surface) |
| Non-root user (`appuser`) | If the app is compromised, attacker has limited permissions |
| `--no-cache-dir` | Don't store pip cache (smaller image) |
| `HEALTHCHECK` | Docker monitors app health automatically |
| `gunicorn` | Production WSGI server (not Flask dev server!) |

### Step 2.1 — The Complete CI/CD Workflow

**`.github/workflows/ci-cd.yaml`:**

```yaml
# .github/workflows/ci-cd.yaml
name: CI/CD Pipeline

# ════════════════════════════════════════════════════
# TRIGGER: When does this pipeline run?
# ════════════════════════════════════════════════════
on:
  push:
    branches: [main, develop]        # Run on push to main or develop
  pull_request:
    branches: [main]                 # Run on PRs targeting main

# ════════════════════════════════════════════════════
# ENVIRONMENT VARIABLES (shared across all jobs)
# ════════════════════════════════════════════════════
env:
  REGISTRY: ghcr.io                                    # GitHub Container Registry
  IMAGE_NAME: ${{ github.repository }}                 # e.g., username/my-production-app
  PYTHON_VERSION: "3.11"

# ════════════════════════════════════════════════════
# JOBS: The pipeline stages
# ════════════════════════════════════════════════════
jobs:

  # ──────────────────────────────────────────────────
  # JOB 1: LINT — Check code quality
  # ──────────────────────────────────────────────────
  lint:
    name: "🔍 Lint Code"
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4
        # ↑ Downloads your code into the runner

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install dependencies
        run: pip install -r requirements-dev.txt

      - name: Run flake8 (linting)
        run: |
          echo "=== Running Flake8 Linter ==="
          flake8 app/ tests/ --count --show-source --statistics
          echo "✅ Linting passed!"
        # flake8 checks for:
        # - Syntax errors
        # - Style violations (PEP 8)
        # - Undefined variables
        # - Unused imports

  # ──────────────────────────────────────────────────
  # JOB 2: TEST — Run unit tests
  # ──────────────────────────────────────────────────
  test:
    name: "🧪 Run Tests"
    runs-on: ubuntu-latest
    needs: lint                      # Only run if lint passes

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install dependencies
        run: pip install -r requirements-dev.txt

      - name: Run tests with coverage
        run: |
          echo "=== Running Tests with Coverage ==="
          coverage run -m pytest tests/ -v --tb=short
          coverage report --fail-under=80
          coverage xml -o coverage.xml
          echo "✅ All tests passed with sufficient coverage!"
        # --fail-under=80: Pipeline fails if test coverage is below 80%
        # -v: Verbose output (shows each test)
        # --tb=short: Short traceback on failures

      - name: Upload coverage report
        uses: actions/upload-artifact@v4
        with:
          name: coverage-report
          path: coverage.xml
        # ↑ Stores the coverage report as a pipeline artifact
        #   You can download it later for review

  # ──────────────────────────────────────────────────
  # JOB 3: SECURITY SCAN — Find vulnerabilities
  # ──────────────────────────────────────────────────
  security-scan:
    name: "🔒 Security Scan"
    runs-on: ubuntu-latest
    needs: lint                      # Run in parallel with tests

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install dependencies
        run: pip install -r requirements-dev.txt

      - name: Run Bandit (SAST — Static Application Security Testing)
        run: |
          echo "=== Running Bandit Security Scanner ==="
          bandit -r app/ -f json -o bandit-report.json --severity-level medium || true
          bandit -r app/ --severity-level medium
          echo "✅ SAST scan complete!"
        # Bandit checks Python code for:
        # - Hardcoded passwords
        # - SQL injection vulnerabilities
        # - Use of insecure functions
        # - Weak cryptography
        # - Command injection risks

      - name: Run Safety (SCA — Software Composition Analysis)
        run: |
          echo "=== Running Safety Dependency Scanner ==="
          safety check -r requirements.txt --output json > safety-report.json || true
          safety check -r requirements.txt
          echo "✅ SCA scan complete!"
        # Safety checks if any of your dependencies have KNOWN vulnerabilities
        # Example: If flask 2.0.0 has a CVE, Safety will flag it

      - name: Upload security reports
        uses: actions/upload-artifact@v4
        with:
          name: security-reports
          path: |
            bandit-report.json
            safety-report.json

  # ──────────────────────────────────────────────────
  # JOB 4: BUILD & PUSH — Create Docker image
  # ──────────────────────────────────────────────────
  build-and-push:
    name: "🐳 Build & Push Image"
    runs-on: ubuntu-latest
    needs: [test, security-scan]     # Only build if tests AND security scans pass
    if: github.ref == 'refs/heads/main'   # Only push images from main branch

    permissions:
      contents: read
      packages: write                # Permission to push to GHCR

    outputs:
      image-tag: ${{ steps.meta.outputs.tags }}
      image-digest: ${{ steps.build.outputs.digest }}

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3
        # ↑ Buildx enables multi-platform builds and advanced caching

      - name: Login to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}
        # ↑ Uses the automatic GITHUB_TOKEN — no manual secret needed!

      - name: Generate image metadata (tags & labels)
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
          tags: |
            type=sha,prefix=                          
            type=raw,value=latest
            type=ref,event=branch
        # This generates tags like:
        # - ghcr.io/username/my-app:abc1234  (git commit SHA)
        # - ghcr.io/username/my-app:latest
        # - ghcr.io/username/my-app:main     (branch name)

      - name: Build and push Docker image
        id: build
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha                        # Use GitHub Actions cache
          cache-to: type=gha,mode=max                 # Save cache for next build
        # Docker layer caching means:
        # - If only app code changed, pip install is CACHED
        # - Build goes from 5 minutes → 30 seconds!

  # ──────────────────────────────────────────────────
  # JOB 5: IMAGE SCAN — Scan the built image
  # ──────────────────────────────────────────────────
  image-scan:
    name: "🔍 Scan Docker Image"
    runs-on: ubuntu-latest
    needs: build-and-push

    steps:
      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest
          format: "table"
          exit-code: "1"             # Fail the pipeline if HIGH/CRITICAL vulns found
          severity: "HIGH,CRITICAL"
          ignore-unfixed: true       # Ignore vulns with no fix available
        # Trivy scans:
        # - OS packages (e.g., vulnerable openssl in the base image)
        # - Application dependencies (e.g., CVE in Flask)
        # - Misconfigurations (e.g., running as root)

      - name: Run Trivy and save report
        uses: aquasecurity/trivy-action@master
        if: always()                 # Run even if previous step failed
        with:
          image-ref: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest
          format: "json"
          output: "trivy-report.json"

      - name: Upload Trivy report
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: trivy-report
          path: trivy-report.json

  # ──────────────────────────────────────────────────
  # JOB 6: UPDATE GITOPS REPO — Trigger deployment
  # ──────────────────────────────────────────────────
  update-gitops:
    name: "🚀 Update GitOps Repo"
    runs-on: ubuntu-latest
    needs: image-scan                # Only deploy if image scan passes

    steps:
      - name: Checkout GitOps repository
        uses: actions/checkout@v4
        with:
          repository: YOUR_USERNAME/gitops-demo    # Your separate GitOps repo
          token: ${{ secrets.GITOPS_PAT }}         # Personal Access Token
          path: gitops-repo

      - name: Update image tag in GitOps repo
        run: |
          cd gitops-repo
          
          # Get the new image tag (git SHA)
          NEW_TAG="${{ github.sha }}"
          NEW_IMAGE="${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${NEW_TAG:0:7}"
          
          echo "=== Updating image to: $NEW_IMAGE ==="
          
          # Update the image in the base deployment using sed
          sed -i "s|image: .*|image: ${NEW_IMAGE}|g" manifests/base/deployment.yaml
          
          # Also update the APP_VERSION in the ConfigMap
          sed -i "s|APP_VERSION:.*|APP_VERSION: \"${NEW_TAG:0:7}\"|g" manifests/base/configmap.yaml
          
          echo "✅ Manifests updated!"
          cat manifests/base/deployment.yaml

      - name: Commit and push changes
        run: |
          cd gitops-repo
          git config user.name "GitHub Actions Bot"
          git config user.email "actions@github.com"
          
          git add .
          git diff --staged --quiet || git commit -m "Deploy image ${{ github.sha }} from CI pipeline

          Triggered by: ${{ github.event_name }}
          Commit: ${{ github.sha }}
          Author: ${{ github.actor }}
          Pipeline: ${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}"
          
          git push
        # This push updates the GitOps repo
        # ArgoCD watches this repo and auto-deploys!
```

Update the GitOps repo's base deployment for our Python app:

**`manifests/base/deployment.yaml`:**

```yaml
# manifests/base/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: my-production-app
  labels:
    app: my-production-app
    version: latest
spec:
  replicas: 1
  selector:
    matchLabels:
      app: my-production-app
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1               # Create 1 extra pod during update
      maxUnavailable: 0         # Never have 0 running pods during update
  template:
    metadata:
      labels:
        app: my-production-app
      annotations:
        prometheus.io/scrape: "true"        # Tell Prometheus to scrape this pod
        prometheus.io/port: "5000"
        prometheus.io/path: "/metrics"
    spec:
      containers:
        - name: app
          image: ghcr.io/YOUR_USERNAME/my-production-app:latest
          ports:
            - containerPort: 5000
              name: http
          envFrom:
            - configMapRef:
                name: app-config
          resources:
            requests:
              cpu: "100m"
              memory: "128Mi"
            limits:
              cpu: "500m"
              memory: "256Mi"
          livenessProbe:
            httpGet:
              path: /health
              port: 5000
            initialDelaySeconds: 10
            periodSeconds: 15
            timeoutSeconds: 5
            failureThreshold: 3
          readinessProbe:
            httpGet:
              path: /ready
              port: 5000
            initialDelaySeconds: 5
            periodSeconds: 10
            timeoutSeconds: 3
            failureThreshold: 3
          startupProbe:
            httpGet:
              path: /health
              port: 5000
            failureThreshold: 30
            periodSeconds: 2
          # Graceful shutdown — let existing requests complete
          lifecycle:
            preStop:
              exec:
                command: ["/bin/sh", "-c", "sleep 5"]
      terminationGracePeriodSeconds: 30
```

**Key production features explained:**

| Feature | What It Does | Why It's Important |
|---------|-------------|-------------------|
| `RollingUpdate` strategy | Updates pods one by one | Zero downtime deployments |
| `maxUnavailable: 0` | Never kills a pod before a new one is ready | Users never see errors |
| `startupProbe` | Gives slow-starting apps extra time | Prevents premature restarts |
| `preStop` lifecycle hook | Waits 5s before stopping | Lets load balancer drain connections |
| `terminationGracePeriodSeconds` | 30s for graceful shutdown | Gunicorn finishes active requests |
| Prometheus annotations | Auto-scraped by Prometheus | Automatic monitoring |

# app/main.py
import logging
from flask import Flask
from prometheus_flask_instrumentator import FlaskInstrumentator
from app.config import Config
from app.routes.health import health_bp
from app.routes.items import items_bp


def create_app():
    """Application factory — creates and configures the Flask app."""
    app = Flask(__name__)

    # Configure logging
    logging.basicConfig(
        level=getattr(logging, Config.LOG_LEVEL),
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s"

    app.register_blueprint(health_bp)
    app.register_blueprint(items_bp, url_prefix="/api")

    FlaskInstrumentator().instrument(app)

    # - http_requests_total (total number of requests)
    # - http_request_size_bytes (request payload size)
    # - http_response_size_bytes (response payload size)
    # All available at /metrics endpoint

    return app


if __name__ == "__main__":

### Step 4.6 — AlertManager Rules

```yaml
# manifests/base/prometheus-rules.yaml
apiVersion: monitoring.coreos.com/v1
kind: PrometheusRule
metadata:
  name: my-production-app-alerts
  labels:
    release: prometheus
spec:
  groups:
    - name: my-production-app
      rules:
        # Alert if error rate is above 5% for 5 minutes
        - alert: HighErrorRate
          expr: |
            sum(rate(http_requests_total{status=~"5..", app="my-production-app"}[5m]))
            /
            sum(rate(http_requests_total{app="my-production-app"}[5m]))
            > 0.05
          for: 5m
          labels:
            severity: critical
          annotations:
            summary: "High error rate detected"
            description: "Error rate is {{ $value | humanizePercentage }} for my-production-app"

        # Alert if response time is above 1 second
        - alert: HighLatency
          expr: |
            histogram_quantile(0.95, 
              rate(http_request_duration_seconds_bucket{app="my-production-app"}[5m])
            ) > 1
          for: 5m
          labels:
            severity: warning
          annotations:
            summary: "High latency detected"
            description: "95th percentile latency is {{ $value }}s"

        # Alert if pod restarts
        - alert: PodRestarting
          expr: |
            increase(kube_pod_container_status_restarts_total{
              pod=~"my-production-app.*"
            }[1h]) > 3
          labels:
            severity: warning
          annotations:
            summary: "Pod is restarting frequently"
            description: "Pod {{ $labels.pod }} has restarted {{ $value }} times in the last hour"

        # Alert if no pods are running
        - alert: NoPodRunning
          expr: |
            kube_deployment_status_replicas_available{
              deployment="my-production-app"
            } == 0
          for: 1m
          labels:
            severity: critical
          annotations:
            summary: "No pods running!"
            description: "my-production-app has 0 available replicas"
```
